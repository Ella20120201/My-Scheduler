        function timeToMinutes(timeStr) {

            const [hours, mins] = timeStr.split(':').map(Number);

            return hours * 60 + mins;

        }


        function minutesToTime(minutes) {

            const hours = Math.floor(minutes / 60) % 24;

            const mins = minutes % 60;

            return `${String(hours).padStart(2, '0')}:${String(mins).padStart(2, '0')}`;

        }


        function generateDailySchedule() {

            if (dailyTasks.length === 0) {

                alert("Please add tasks to generate a schedule plan.");

                return;

            }


            const startTimeVal = document.getElementById('startTime').value;

            const endTimeVal = document.getElementById('endTime').value;

            const breakDuration = parseInt(document.getElementById('breakDuration').value) || 0;

            const style = document.getElementById('workStyle').value;


            let currentMinutes = timeToMinutes(startTimeVal);

            const limitMinutes = timeToMinutes(endTimeVal);


            let sortedTasks = [...dailyTasks];


            if (style === 'frog') {

                const priorityWeight = { high: 3, medium: 2, low: 1 };

                sortedTasks.sort((a, b) => {

                    if (priorityWeight[b.priority] !== priorityWeight[a.priority]) {

                        return priorityWeight[b.priority] - priorityWeight[a.priority];

                    }

                    return b.duration - a.duration;

                });

            } else if (style === 'quick') {

                sortedTasks.sort((a, b) => a.duration - b.duration);

            } else {

                const priorityWeight = { high: 3, medium: 2, low: 1 };

                sortedTasks.sort((a, b) => priorityWeight[b.priority] - priorityWeight[a.priority]);

            }


            let timelineHtml = '';

            let totalWork = 0;

            let totalBreak = 0;

            let finalTime = currentMinutes;


            for (let i = 0; i < sortedTasks.length; i++) {

                const task = sortedTasks[i];

                const taskStart = currentMinutes;

                const taskEnd = currentMinutes + task.duration;

                const isOverflowing = taskEnd > limitMinutes;


                let priorityBorder = 'border-l-4 border-l-slate-400';

                if (task.priority === 'high') priorityBorder = 'border-l-4 border-l-red-500';

                if (task.priority === 'medium') priorityBorder = 'border-l-4 border-l-yellow-500';

                if (task.priority === 'low') priorityBorder = 'border-l-4 border-l-green-500';


                timelineHtml += `

                    <div class="p-4 bg-white rounded-lg border border-slate-200 shadow-xs ${priorityBorder} ${isOverflowing ? 'bg-red-50/40 border-red-200' : ''}">

                        <div class="flex justify-between items-start mb-1">

                            <span class="text-xs font-semibold text-indigo-600">${minutesToTime(taskStart)} - ${minutesToTime(taskEnd)}</span>

                            <span class="text-xs font-medium text-slate-400">${task.duration} mins</span>

                        </div>

                        <h3 class="font-medium text-slate-900 text-sm">${task.name}</h3>

                        ${isOverflowing ? `<p class="text-red-500 text-[10px] font-medium mt-1">⚠️ Out of schedule boundary</p>` : ''}

                    </div>

                `;


                totalWork += task.duration;

                currentMinutes = taskEnd;

                finalTime = currentMinutes;


                if (i < sortedTasks.length - 1 && breakDuration > 0) {

                    const breakStart = currentMinutes;

                    const breakEnd = currentMinutes + breakDuration;

                    const isBreakOverflowing = breakEnd > limitMinutes;


                    timelineHtml += `

                        <div class="flex items-center space-x-4 px-4 py-1.5 border-l-4 border-l-sky-300 bg-sky-50/50 rounded-lg text-xs ${isBreakOverflowing ? 'opacity-70' : ''}">

                            <span class="font-semibold text-sky-700">${minutesToTime(breakStart)} - ${minutesToTime(breakEnd)}</span>

                            <span class="text-slate-600 font-medium">☕ Short Break (${breakDuration} mins)</span>

                        </div>

                    `;


                    totalBreak += breakDuration;

                    currentMinutes = breakEnd;

                    finalTime = currentMinutes;

                }

            }


            document.getElementById('scheduleTimeline').innerHTML = timelineHtml;

            document.getElementById('totalWorkTime').textContent = `${totalWork}m`;

            document.getElementById('totalBreakTime').textContent = `${totalBreak}m`;

            document.getElementById('actualEndTime').textContent = minutesToTime(finalTime);

            

            if (finalTime > limitMinutes) {

                document.getElementById('actualEndTime').className = "text-base font-bold text-red-500";

            } else {

                document.getElementById('actualEndTime').className = "text-base font-bold text-emerald-600";

            }

            document.getElementById('scheduleSummary').classList.remove('hidden');

        }

        function addWeeklyTask(event) {

            event.preventDefault();

            const day = document.getElementById('weeklyDaySelect').value;

            const taskText = document.getElementById('weeklyTaskName').value.trim();

            document.getElementById('weeklyTaskName').value = '';

            

            renderWeeklyPlan();

        function deleteWeeklyTask(day, id) {

            weeklyTasks[day] = weeklyTasks[day].filter(t => t.id !== id);

            localStorage.setItem('weeklyTasks', JSON.stringify(weeklyTasks));

            renderWeeklyPlan();

        }


        function toggleWeeklyTask(day, id) {

            const task = weeklyTasks[day].find(t => t.id === id);

            if (task) {

                task.completed = !task.completed;

                localStorage.setItem('weeklyTasks', JSON.stringify(weeklyTasks));

                renderWeeklyPlan();

            }

        }

                            <div class="flex items-start space-x-1.5 flex-grow pr-1">

                        </div>

                    `;

        }

        function renderCalendar() {

            const gridEl = document.getElementById('calendarGrid');

            const titleEl = document.getElementById('calendarTitle');


            const year = currentViewDate.getFullYear();

            const month = currentViewDate.getMonth();


            const months = ["January", "February", "March", "April", "May", "June", "July", "August", "September", "October", "November", "December"];

            titleEl.textContent = `${months[month]} ${year}`;


            // Helper formulas for calendar month structure

            const firstDayOfMonth = new Date(year, month, 1);

            let firstDayIndex = firstDayOfMonth.getDay(); // 0 = Sun, 1 = Mon ...

            let startBlankDays = firstDayIndex === 0 ? 6 : firstDayIndex - 1; // Align Mon as index 0


            const totalDays = new Date(year, month + 1, 0).getDate();


            let cellsHtml = '';


            // Blank spacer cells

            for (let i = 0; i < startBlankDays; i++) {

                cellsHtml += `<div class="aspect-square bg-slate-50 border border-slate-100/50 rounded-lg"></div>`;

            }


            // Calendar active date loops

            for (let d = 1; d <= totalDays; d++) {

                const dateStr = `${year}-${String(month + 1).padStart(2, '0')}-${String(d).padStart(2, '0')}`;

                const logged = efficiencyLogs[dateStr];


                let colorClass = 'bg-slate-100 hover:bg-slate-200 text-slate-700';

                if (logged) {

                    if (logged.score === 1) colorClass = 'bg-indigo-100 text-indigo-700 border border-indigo-200 font-bold';

                    if (logged.score === 2) colorClass = 'bg-indigo-200 text-indigo-800 border border-indigo-300 font-bold';

                    if (logged.score === 3) colorClass = 'bg-indigo-300 text-indigo-900 border border-indigo-400 font-bold';

                    if (logged.score === 4) colorClass = 'bg-indigo-500 text-white font-bold';

                    if (logged.score === 5) colorClass = 'bg-indigo-700 text-white font-bold';

                }


                // Check if current cell date is today

                const todayStr = new Date().toISOString().split('T')[0];

                const isToday = dateStr === todayStr;

                const ringClass = isToday ? 'ring-2 ring-emerald-500 ring-offset-1' : '';


                cellsHtml += `

                    <button onclick="selectDateToLog('${dateStr}')" 

                        class="aspect-square p-1 rounded-lg text-xs flex flex-col justify-between items-center transition relative group ${colorClass} ${ringClass}">

                        <span>${d}</span>

                        ${logged ? `<span class="text-[9px] scale-75">★${logged.score}</span>` : ''}

                        

                        <!-- Tooltip on hover -->

                        ${logged ? `

                        <span class="absolute z-10 hidden group-hover:block bg-slate-900 text-white text-[10px] p-2 rounded shadow-lg -top-12 left-1/2 transform -translate-x-1/2 w-32 font-normal leading-normal">

                            Score: ${logged.score}/5<br>${logged.note ? logged.note.substring(0,25)+'...' : 'No notes'}

                        </span>` : ''}

                    </button>

                `;

            }


            gridEl.innerHTML = cellsHtml;

        }


        function renderEfficiencyLogList() {

            const listEl = document.getElementById('efficiencyLogsList');

            const sortedDates = Object.keys(efficiencyLogs).sort((a,b) => new Date(b) - new Date(a));


            if (sortedDates.length === 0) {

                listEl.innerHTML = `<p class="text-sm text-slate-400 text-center py-4">No logged statistics yet. Rate a day using the form to populate.</p>`;

                return;

            }


            listEl.innerHTML = sortedDates.map(date => {

                const log = efficiencyLogs[date];

                let stars = '⭐'.repeat(log.score);

                

                return `

                    <div class="p-3 bg-slate-50 border border-slate-200 rounded-lg text-xs flex justify-between items-start">

                        <div class="space-y-1">

                            <div class="flex items-center space-x-2">

                                <span class="font-bold text-slate-700">${date}</span>

                                <span class="text-amber-500">${stars}</span>

                            </div>

                            <p class="text-slate-600 leading-normal">${log.note || '<span class="italic text-slate-400">No notes written</span>'}</p>

                        </div>

                        <button onclick="deleteEfficiencyLog('${date}')" class="text-slate-300 hover:text-red-500 font-bold px-1 text-sm">×</button>

                    </div>

                `;

            }).join('');

        }
